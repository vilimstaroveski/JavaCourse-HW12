13  import javax.swing.AbstractAction;
14  import javax.swing.JFileChooser;
15  import javax.swing.JOptionPane;
16  /**
17   * Action that saves current picture as in .jvd format
18   * @author Vilim StaroveÅ¡ki
19   *
20   */
21  public class SaveAsAction extends AbstractAction {
22  
23  	/**Default serial version UID. */
24  	private static final long serialVersionUID = 1L;
25  	/**{@link JVDraw} in which this action is performed in*/
26  	private JVDraw parent;
27  	/**
28  	 * Creates new {@link SaveAsAction}
29  	 * @param jvDraw {@link JVDraw} in which this action is performed in
30  	 */
31  	public SaveAsAction(JVDraw jvDraw) {
32  		parent = jvDraw;
33  	}
34  
35  	@Override
36  	public void actionPerformed(ActionEvent e) {
37  
38  		JFileChooser fc = new JFileChooser();
39  		fc.setDialogTitle("Save document"); 
40  		if(fc.showSaveDialog(parent) != JFileChooser.APPROVE_OPTION) {
41  			JOptionPane.showMessageDialog(parent, 
42  					"Nothing has been saved!", 
43  					"System messege", 
44  					JOptionPane.INFORMATION_MESSAGE);
45  			return;
46  		}
47  		Path file = fc.getSelectedFile().toPath();
48  		if(!file.toString().endsWith("jvd")) {
49  			file = Paths.get(file.toString()+".jvd");
50  		}
51  		if(Files.exists(file)) {
52  			int r = JOptionPane.showConfirmDialog(parent, 
53  					"File "+file+" already exist. Do you want to overwrite it?", 
54  					"Warning", 
55  					JOptionPane.YES_NO_OPTION,
56  					JOptionPane.QUESTION_MESSAGE);
57  			if(r != JOptionPane.YES_OPTION) {
58  				return;
59  			}
60  		}
61  		parent.setPathToCurrentFile(file);
62  		
63  		String modelText = "";
64  		for(int i = 0; i < parent.getDrawingModel().getSize(); i++) {
65  			GeometricalObject object = parent.getDrawingModel().getObject(i);
66  			modelText += object.toString()+"\r\n";
67  		}
68  		
69  		try {
70  			Files.write(parent.getPathToCurrentFile(), modelText.getBytes(StandardCharsets.UTF_8));
71  		} catch (IOException e1) {
72  			JOptionPane.showMessageDialog(parent, 
73  					e1.getMessage(),
74  					"Error", 
75  					JOptionPane.ERROR_MESSAGE);
76  			return;
77  		}
78  		parent.setModified(false);
79  	}
80  
81  }
