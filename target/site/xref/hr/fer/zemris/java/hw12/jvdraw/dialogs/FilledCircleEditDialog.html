8   import javax.swing.JLabel;

10  import javax.swing.JTextField;
11  /**
12   * Edit dialog for editing filled circle. It can edit filled circles 
13   * center point, radius, outline color and inside color.
14   * @author Vilim StaroveÅ¡ki
15   *
16   */
17  public class FilledCircleEditDialog extends JPanel{
18  
19  	/**Default serial version UID*/
20  	private static final long serialVersionUID = 1L;
21  	/**Filled circle to edit*/
22  	private FilledCircle fcircle;
23  	/**Area for definig center point's x coordinate*/
24  	private JTextField startX;
25  	/**Area for definig center point's y coordinate*/
26  	private JTextField startY;
27  	/**Area for definig circles radius*/
28  	private JTextField radius;
29  	/**Area for definig circles line color*/
30  	private JColorArea outlineColors;
31  	/**Area for definig circles inside color*/
32  	private JColorArea insideColors;
33  	
34  	/**
35  	 * Creates new {@link CircleEditDialog}
36  	 * @param fcircle circle that will be redefined
37  	 */
38  	public FilledCircleEditDialog(FilledCircle fcircle) {
39  		this.fcircle = fcircle;
40  		initDialog();
41  		
42  	}
43  	/**
44  	 * Initializes dialogs GUI.
45  	 */
46  	private void initDialog() {
47  		setLayout(new GridLayout(5, 2));
48  		//row 1
49  		add(new JLabel("Center point X:"));
50  		startX = new JTextField(Integer.toString(fcircle.getStartPointX()));
51  		add(startX);
52  		//row 2
53  		add(new JLabel("Center point Y:"));
54  		startY = new JTextField(Integer.toString(fcircle.getStartPointY()));
55  		add(startY);
56  		//row 3 
57  		add(new JLabel("Radius:"));
58  		radius = new JTextField(Integer.toString(fcircle.getRadius()));
59  		add(radius);
60  		//row 4
61  		add(new JLabel("Outline color"));
62  		outlineColors = new JColorArea(fcircle.getOutline());
63  		add(outlineColors);
64  		//row 5
65  		add(new JLabel("Inside color"));
66  		insideColors = new JColorArea(fcircle.getInside());
67  		add(insideColors);
68  		
69  	}
70  	/**
71  	 * Performes all defined changes on circle.
72  	 */
73  	public void updateObject() {
74  		fcircle.setStartPoint(Integer.parseInt(startX.getText()), Integer.parseInt(startY.getText()));
75  		fcircle.setEndPoint(Integer.parseInt(startX.getText())+Integer.parseInt(radius.getText()), Integer.parseInt(startY.getText()));
76  		fcircle.setOutline(outlineColors.getCurrentColor());
77  		fcircle.setInsideColor(insideColors.getCurrentColor());
78  	}
79  }
