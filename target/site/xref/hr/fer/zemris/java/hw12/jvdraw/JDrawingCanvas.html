19  /**
20   * Component on which objects will be renderd on.
21   * @author Vilim StaroveÅ¡ki
22   *
23   */
24  public class JDrawingCanvas extends JComponent implements DrawingModelListener {
25  	
26  	/**Default serial version UID*/
27  	private static final long serialVersionUID = 1L;
28  	/**Model containing all objects present on picture*/
29  	private DrawingModel model;
30  	/**Currently drawn object.*/
31  	private GeometricalObject currentlyDrawing;
32  	/**Group of draw buttons*/
33  	private ButtonGroup group;
34  	/**Foreground color of objects*/
35  	private JColorArea foreground;
36  	/**Background color of objects*/
37  	private JColorArea background;
38  	/**{@link JVDraw} in which this canvas is located*/
39  	private JVDraw parent;
40  	/**
41  	 * Creates new {@link JDrawingCanvas}.
42  	 * @param parent {@link JVDraw} in which this canvas is located
43  	 */
44  	public JDrawingCanvas(JVDraw parent) {
45  		this.model = parent.getDrawingModel();
46  		this.group = parent.getDrawButtons();
47  		this.foreground = parent.getForeGround();
48  		this.background = parent.getBackGround();
49  		this.parent = parent;
50  
51  		currentlyDrawing = null;
52  		MyMouseAdapter mouseListener = new MyMouseAdapter();
53  		this.addMouseListener(mouseListener);
54  		this.addMouseMotionListener(mouseListener);
55  	}
56  
57  	@Override
58  	public void objectsAdded(DrawingModel source, int index0, int index1) {
59  
60  		parent.setModified(true);
61  		repaint();
62  	}
63  
64  	@Override
65  	public void objectsRemoved(DrawingModel source, int index0, int index1) {
66  		
67  		parent.setModified(true);
68  		repaint();
69  	}
70  
71  	@Override
72  	public void objectsChanged(DrawingModel source, int index0, int index1) {
73  		
74  		parent.setModified(true);
75  		repaint();
76  	}

78  	/**
79  	 * Private class which defines mouse actions on this canvas.
80  	 * @author Vilim StaroveÅ¡ki
81  	 *
82  	 */
83  	private class MyMouseAdapter extends MouseAdapter {
84  		
85  		@Override
86  		public void mouseMoved(MouseEvent e) {
87  			super.mouseMoved(e);
88  			if(currentlyDrawing != null) {
89  				currentlyDrawing.setEndPoint(e.getPoint().x, e.getPoint().y);
90  				repaint();
91  			}
92  		}
93  		
94  		@Override
95  		public void mouseClicked(MouseEvent e) {
96  			super.mouseClicked(e);
97  			if(currentlyDrawing == null) {
98  				if(group.getSelection().getActionCommand().equals("LINE")) {
99  					currentlyDrawing = new Line(foreground.getCurrentColor());
100 				}
101 				else if(group.getSelection().getActionCommand().equals("CIRCLE")) {
102 					currentlyDrawing = new Circle(foreground.getCurrentColor());
103 				}
104 				else if(group.getSelection().getActionCommand().equals("FCIRCLE")) {
105 					currentlyDrawing = new FilledCircle(foreground.getCurrentColor(), background.getCurrentColor());
106 				}
107 				currentlyDrawing.setStartPoint(e.getPoint().x, e.getPoint().y);
108 			}
109 			else {
110 				currentlyDrawing.setEndPoint(e.getPoint().x, e.getPoint().y);
111 				model.add(currentlyDrawing);
112 				currentlyDrawing = null;
113 			}
114 		}
115 	}
116 	
117 	@Override
118 	protected void paintComponent(Graphics g) {
119 		super.paintComponent(g);
120 		Graphics2D g2d = (Graphics2D) g;
121 		
122 		g2d.setColor(Color.WHITE);
123 		g2d.fillRect(0, 0, JDrawingCanvas.this.getWidth(), JDrawingCanvas.this.getHeight());
124 		
125 		if(currentlyDrawing != null) {
126 			if(currentlyDrawing instanceof Line) {
127 				
128 				g2d.setColor(currentlyDrawing.getOutline());
129 				g2d.drawLine(currentlyDrawing.getStartPointX(), 
130 						currentlyDrawing.getStartPointY(), 
131 						currentlyDrawing.getEndPointX(), 
132 						currentlyDrawing.getEndPointY());
133 			}
134 			if(currentlyDrawing instanceof Circle) {
135 				
136 				int diammeter = 2 * ( (Circle)currentlyDrawing).getRadius();
137 				g2d.setColor(currentlyDrawing.getOutline());
138 				g2d.drawOval(currentlyDrawing.getStartPointX()-((Circle)currentlyDrawing).getRadius(), 
139 						currentlyDrawing.getStartPointY()-((Circle)currentlyDrawing).getRadius(), 
140 						diammeter, 
141 						diammeter);
142 				if(currentlyDrawing instanceof FilledCircle) {
143 					g2d.setColor( ((FilledCircle)currentlyDrawing).getInside());
144 					g2d.fillOval(currentlyDrawing.getStartPointX()-((Circle)currentlyDrawing).getRadius(), 
145 						currentlyDrawing.getStartPointY()-((Circle)currentlyDrawing).getRadius(),
146 						diammeter, 
147 						diammeter);
148 				}
149 			}
150 		}
151 		
152 		for(int i = 0; i < model.getSize(); i++) {
153 			GeometricalObject object = model.getObject(i);
154 			if(object instanceof Line) {
155 				
156 				g2d.setColor(object.getOutline());
157 				g2d.drawLine(object.getStartPointX(), 
158 						object.getStartPointY(), 
159 						object.getEndPointX(), 
160 						object.getEndPointY());
161 			}
162 			if(object instanceof Circle) {
163 				
164 				int diammeter = 2 * ( (Circle)object).getRadius();
165 				g2d.setColor(object.getOutline());
166 				g2d.drawOval(object.getStartPointX()-((Circle)object).getRadius(), 
167 						object.getStartPointY()-((Circle)object).getRadius(),
168 						diammeter, 
169 						diammeter);
170 				if(object instanceof FilledCircle) {
171 					g2d.setColor( ((FilledCircle)object).getInside());
172 					g2d.fillOval(object.getStartPointX()-((Circle)object).getRadius(), 
173 						object.getStartPointY()-((Circle)object).getRadius(),
174 						diammeter, 
175 						diammeter);
176 				}
177 			}
178 		}
179 	}
