6   import java.awt.Dimension;
7   import java.awt.GridLayout;
8   
9   import javax.swing.JLabel;
10  import javax.swing.JPanel;
11  import javax.swing.JTextField;
12  /**
13   * Edit dialog for editing line. It can edit lines 
14   * start and end point and line color
15   * @author Vilim StaroveÅ¡ki
16   *
17   */
18  public class LineEditDialog extends JPanel {
19  	
20  	/**Default serial version UID*/
21  	private static final long serialVersionUID = 1L;
22  	/**Line that is going to be modified*/
23  	private Line line;
24  	/**Area for defining line's start point x coordinate*/
25  	private JTextField startX;
26  	/**Area for defining line's start point y coordinate*/
27  	private JTextField startY;
28  	/**Area for defining line's end point x coordinate*/
29  	private JTextField endX;
30  	/**Area for defining line's end point y coordinate*/
31  	private JTextField endY;
32  	/**Area for defining line's color*/
33  	private JColorArea colors;
34  	/**
35  	 * Creates new {@link LineEditDialog}
36  	 * @param line line to define.
37  	 */
38  	public LineEditDialog(Line line) {
39  		this.line = line;
40  		initDialog();
41  	}
42  	/**
43  	 * Initializes GUI of this dialog
44  	 */
45  	private void initDialog() {
46  		setLayout(new GridLayout(5, 2));
47  
48  		//row 1
49  		add(new JLabel("Start point X:"));
50  		startX = new JTextField(Integer.toString(line.getStartPointX()));
51  		add(startX);
52  		//row 2
53  		add(new JLabel("Start point Y:"));
54  		startY = new JTextField(Integer.toString(line.getStartPointY()));
55  		add(startY);
56  		//row 3 
57  		add(new JLabel("End point X:"));
58  		endX = new JTextField(Integer.toString(line.getEndPointX()));
59  		add(endX);
60  		//row 4
61  		add(new JLabel("End point Y:"));
62  		endY = new JTextField(Integer.toString(line.getEndPointY()));
63  		add(endY);
64  		//row 5
65  		add(new JLabel("Line color"));
66  		colors = new JColorArea(line.getOutline());
67  		colors.setMaximumSize(new Dimension(15, 15));
68  		add(colors);
69  	}
70  	/**
71  	 * Performes given modifications on line
72  	 */
73  	public void updateObject() {
74  		line.setStartPoint(Integer.parseInt(startX.getText()), Integer.parseInt(startY.getText()));
75  		line.setEndPoint(Integer.parseInt(endX.getText()), Integer.parseInt(endY.getText()));
76  		line.setOutline(colors.getCurrentColor());
77  	}
78  
79  }
