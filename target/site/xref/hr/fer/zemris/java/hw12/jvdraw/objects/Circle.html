4   import java.awt.Point;
5   /**
6    * Object represents drawn circle.

10  public class Circle extends GeometricalObject {
11  	/**
12  	 * Creates new {@link Circle}
13  	 * @param outline outline color
14  	 */
15  	public Circle(Color outline) {
16  		super();
17  		this.outline = outline;
18  		name = "CIRCLE";
19  	}
20  	/**
21  	 * Returns circle radius.
22  	 * @return circle radius.
23  	 */
24  	public int getRadius() {
25  		return (int) Math.round( 
26  				Math.sqrt( 
27  						Math.pow(startPointX - endPointX, 2) + 
28  						Math.pow(startPointY - endPointY, 2)
29  						));
30  	}
31  	
32  	@Override
33  	public Point getUpperLeftCorner() {
34  		int radius = getRadius();
35  		return new Point(startPointX-radius, startPointY-radius);
36  	}
37  
38  	@Override
39  	public Point getDownRightCorner() {
40  		int radius = getRadius();
41  		return new Point(startPointX+radius, startPointY+radius);
42  	}
43  	
44  	@Override
45  	public String toString() {
46  		return name+" "+startPointX+
47  				" "+startPointY+
48  				" "+getRadius()+
49  				" "+outline.getRed()+
50  				" "+outline.getGreen()+
51  				" "+outline.getBlue();
52  	}
53  }
