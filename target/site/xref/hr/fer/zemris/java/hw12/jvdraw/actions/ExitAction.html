9   /**
10   * Action that exites the {@link JVDraw} program.
11   * @author Vilim StaroveÅ¡ki
12   *
13   */
14  public class ExitAction extends AbstractAction {
15  
16  	/** Default serial version UID. */
17  	private static final long serialVersionUID = 1L;
18  	/**{@link JVDraw} in which this action is performed*/
19  	private JVDraw parent;
20  	/**
21  	 * Creates new {@link ExitAction}
22  	 * @param jvDraw {@link JVDraw} in which this action is performed
23  	 */
24  	public ExitAction(JVDraw jvDraw) {
25  		parent = jvDraw;
26  	}
27  
28  	@Override
29  	public void actionPerformed(ActionEvent e) {
30  		
31  		if(parent.getModified()) {
32  
33  			int r = JOptionPane
34  					.showConfirmDialog(
35  							parent,
36  							"There are unsaved changes in this file, do you want to save changes?",
37  							"Warning", JOptionPane.YES_NO_CANCEL_OPTION,
38  							JOptionPane.QUESTION_MESSAGE);
39  			if (r == JOptionPane.YES_OPTION) {
40  				parent.save();
41  			}
42  			else if (r == JOptionPane.CANCEL_OPTION) {
43  				return;
44  			}
45  			parent.dispose();
46  		}
47  		else {
48  
49  			parent.dispose();
50  		}
51  	}
52  
53  }
