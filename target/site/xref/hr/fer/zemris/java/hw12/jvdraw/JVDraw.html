121 	 * Initializes GUI of this program.
122 	 */
123 	private void initGUI() {
124 		
125 		getContentPane().setLayout(new BorderLayout());
126 		
127 		drawButtons = new ButtonGroup();
128 		
129 		drawingModel = new ObjectsModel();
130 		DrawingObjectListModel objectList = new DrawingObjectListModel(drawingModel);
131 		listOfObjects = new JList<GeometricalObject>(objectList);
132 		listOfObjects.setBorder(BorderFactory.createRaisedBevelBorder());
133 		listOfObjects.setBackground(Color.LIGHT_GRAY);
134 		drawingModel.addDrawingModelListener(objectList);
135 		listOfObjects.addMouseListener(new MouseAdapter() {
136 			@Override
137 			public void mouseClicked(MouseEvent e) {
138 				super.mouseClicked(e);
139 				if(e.getClickCount() != 2) {
140 					return;
141 				}
142 				int selectedIndex = listOfObjects.getSelectedIndex();
143 				
144 				GeometricalObject object = drawingModel.getObject(selectedIndex);
145 				if(object instanceof Line) {
146 					LineEditDialog dialog = new LineEditDialog((Line)object);
147 					int result = JOptionPane.showConfirmDialog(JVDraw.this, 
148 							dialog, 
149 							"Edit line", 
150 							JOptionPane.OK_CANCEL_OPTION, 
151 							JOptionPane.PLAIN_MESSAGE
152 							);
153 					if(result == JOptionPane.OK_OPTION) {
154 						dialog.updateObject();
155 						canvas.repaint();
156 						modified = true;
157 					}
158 				}
159 				else if(object instanceof FilledCircle) {
160 					FilledCircleEditDialog dialog = new FilledCircleEditDialog((FilledCircle)object);
161 					int result = JOptionPane.showConfirmDialog(JVDraw.this, 
162 							dialog, 
163 							"Edit filled circle", 
164 							JOptionPane.OK_CANCEL_OPTION, 
165 							JOptionPane.PLAIN_MESSAGE
166 							);
167 					if(result == JOptionPane.OK_OPTION) {
168 						dialog.updateObject();
169 						canvas.repaint();
170 						modified = true;
171 					}
172 				}
173 				else if(object instanceof Circle) {
174 					CircleEditDialog dialog = new CircleEditDialog((Circle)object);
175 					int result = JOptionPane.showConfirmDialog(JVDraw.this, 
176 							dialog, 
177 							"Edit circle", 
178 							JOptionPane.OK_CANCEL_OPTION, 
179 							JOptionPane.PLAIN_MESSAGE
180 							);
181 					if(result == JOptionPane.OK_OPTION) {
182 						dialog.updateObject();
183 						canvas.repaint();
184 						modified = true;
185 					}
186 				}
187 				
188 			}
189 		});
190 		createActions();
191         createMenus();
192         createToolBar();
193         
194         canvas = new JDrawingCanvas(JVDraw.this);
195 		canvas.setBorder(BorderFactory.createLoweredBevelBorder());
196 		drawingModel.addDrawingModelListener(canvas);
197 		
198 		getContentPane().add(canvas, BorderLayout.CENTER);
199 		getContentPane().add(listOfObjects, BorderLayout.EAST);
200 		
201 	}

247 	/**
248 	 * Modifies actions so they are user friendly.
249 	 * @param action what action does
250 	 * @param name name of action
251 	 * @param accelerator accelerator key
252 	 * @param mnemonic mnemonic key
253 	 * @param description short description 
254 	 * @param smallIcon small icon for action in menu
255 	 * @param largeIcon large icon for action in toolbar
256 	 */
257 	private void modifyAction(AbstractAction action, String name,
258 			String accelerator, int mnemonic, String description, ImageIcon smallIcon, 
259 			ImageIcon largeIcon) {
260 		
261 		if(name.isEmpty()) {
262 			action.putValue(Action.LARGE_ICON_KEY, largeIcon);
263 		}
264 		else {
265 			action.putValue(Action.NAME, name);
266 		}
267 		
268 		if(!accelerator.isEmpty()) {
269 			action.putValue(Action.ACCELERATOR_KEY, KeyStroke.getKeyStroke(accelerator));
270 		}
271 		if(mnemonic != -1) {
272 			action.putValue(Action.MNEMONIC_KEY, mnemonic);
273 		}
274 		action.putValue(Action.SHORT_DESCRIPTION, description);
275 		action.putValue(Action.SMALL_ICON, smallIcon);
276 		
277 	}

288 		background = new JColorArea(Color.BLUE);
289 		foreground.setToolTipText("Select foreground color.");
290 		background.setToolTipText("Select background color.");
291 		toolBar.add(foreground);
292 		toolBar.add(background);
293 		
294 		toolBar.addSeparator();
295 		
296 		addToolBarButtons(toolBar);
297 		
298 		StatusLabel stats = new StatusLabel(background, foreground);
299 		foreground.addColorChangeListener(stats);
300 		background.addColorChangeListener(stats);
301 		getContentPane().add(stats, BorderLayout.SOUTH);
302 		
303 		getContentPane().add(toolBar, BorderLayout.NORTH);
304 	}

318 		
319 		drawButtons.add(lineButton);
320 		drawButtons.add(circleButton);
321 		drawButtons.add(fcircleButton);
322 		
323 		drawButtons.setSelected(lineButton.getModel(), true);
324 		
325 		toolBar.add(lineButton);
326 		toolBar.add(circleButton);
327 		toolBar.add(fcircleButton);
328 	}
